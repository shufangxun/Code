### **1. 刷题**   

**字符串中第一个出现频率一次的字符**  


常规做法 hash映射
```python
class Solution:
    def firstNotRepeatingChar(self, s):
        """
        :type s: str
        :rtype: str
        """
        d = dict()
        for i in range(len(s)):
            if s[i] not in d.keys():
                d[s[i]] = 1
            else:
                d[s[i]] += 1

        for i in range(len(s)):
            if d[s[i]] == 1:
                return s[i]

        return '#'
```

**字符流中第一个只出现一次的字符**  

```python

class Solution:
    def __init__(self):
        self.queue = []
        self.d = dict()  
    def firstAppearingOnce(self):
        """
        :rtype: str
        """
        for i in range(len(self.queue)):
            if self.d[self.queue[i]] == 1:
                return self.queue[i]

        return '#'
        
    def insert(self, char):
        """
        :type char: str
        :rtype: void
        """
        if char in self.d.keys():
            self.d[char] += 1
        else:
            self.d[char] = 1
            self.queue.append(char)


```
