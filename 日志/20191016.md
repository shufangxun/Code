**0到n-1中缺失的数字**
```python
def getmiss(nums):
    low, high = 0, len(nums) - 1
    if len(nums) < 0:
        return 0
    if nums[high] == high:
        return high + 1
    while low < high:
        mid = low + (high - low) // 2
        if nums[mid] == mid: # 后半部问题
            low = mid + 1
        else:
            high = mid
    return nums[high] - 1
```

def verfypost(sequence):
    if  not sequence:
        return True 
            
    start = 0
    while sequence:
        while sequence[start] < sequence[-1]:
            start += 1
        while sequence[start] > sequence[-1]:
            start += 1

        start = 0

**数组中只出现一次的数**  
第一种: 除了两个出现一次,其余都是出现两次
```python
def findnum(nums):
    tmp = 0
    for num in nums:
        tmp = tmp ^ num
    
    index = 0
    while tmp & 1 == 0:
        tmp = tmp >> 1
        index += 1
    
    a, b = 0, 0
    for num in nums:
        if num >> index & 1 == 0:
            a = a ^ num
        else:
            b = b ^ num
    return [a, b]
```

第二种: 除了一个出现一次,其余都是出现三次
```python
def findnum(nums):
    ones = twos = 0
    for num in nums:
        ones = (ones ^ num) & ~twos
        twos = (twos ^ num) & ~ones
    return ones
```
